# FFT calculation function
def fft_amp(time, data):
    x0 = np.hanning(len(data))
    data = x0 * data
    amp =(2/data.size)*np.fft.fft(data)
    amp = 2 * abs(amp[range(int(len(data) / 2))])
    freq = np.linspace(0, f_sample // 2, np.int(amp.size))
    return freq, amp

def find_nearest(array, value):
    array = np.asarray(array)
    idx = (np.abs(array - value)).argmin()
    return array[idx], idx
